📦 PASO 1: CONSTRUIR Y SUBIR IMAGEN (donde tengas Docker)
═══════════════════════════════════════════════════════════════

# Opción A - Con el script:
git checkout feature/ai-natural-language-interaction
./build-testing.sh

# Opción B - Manual:
docker build -t alvaromolrui/musicalo:testing .
docker login
docker push alvaromolrui/musicalo:testing

═══════════════════════════════════════════════════════════════
🚀 PASO 2: PROBAR EN TU SERVIDOR (MUY SIMPLE)
═══════════════════════════════════════════════════════════════

ssh tu-usuario@tu-servidor
cd /ruta/a/musicalo

# Editar docker-compose.yml
nano docker-compose.yml

# Cambiar esta línea:
  image: alvaromolrui/musicalo:latest
# Por:
  image: alvaromolrui/musicalo:testing

# Guardar (Ctrl+O) y salir (Ctrl+X)

# Actualizar y reiniciar
docker-compose pull
docker-compose down
docker-compose up -d

# Ver logs
docker-compose logs -f
